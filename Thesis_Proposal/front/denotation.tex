\begin{denotation}[3cm]

\item[NA]{
  數值孔徑 (Numerical Aperture)
}

\item[OTF]{
  光學傳遞函數 (Optical Transfer Function)
}

\item[PSF]{
  點擴散函數 (Point Spread Function)
}

\item[SNR]{
  訊雜比 (Signal-to-Noise Ratio)
}

\item[FP]{
  傅立葉疊層顯微術 (Fourier Ptychography)
}

\item[piFP]{
  圖案化照明傅立葉疊層成像 (Pattern-Illuminated Fourier Ptychography)
}

\item[ePIE]{
  擴展型疊代相位復原引擎 (Extended Ptychographical Iterative Engine)
}

\item[SSE]{
  總平方誤差 (Sum Square Error)
}

\item[FFT]{
  快速傅立葉轉換 (Fast Fourier Transform)
}

\item[IFFT]{
  快速反傅立葉轉換 (Inverse Fast Fourier Transform)
}

\item[VR]{
  震動衰減 (Vibrational Relaxation)
}

\item[ISC]{
  系間跨越 (Intersystem Crossing)
}

\item[$\lambda$]{
  照明波長 (Wavelength of illumination)
}

\item[$\lambda_{ex}$]{
  激發光波長 (Excitation wavelength)
}

\item[$\lambda_{em}$]{
  發射光波長 (Emission wavelength)
}

\item[$h$]{
  普朗克常數 (Planck's constant)
}

\item[$\mathcal{F}$]{
  傅立葉轉換運算子 (Fourier transform operator)
}

\item[$\mathcal{P}$]{
  菲涅耳衍射傳播運算子 (Fresnel propagation operator)
}

\item[$\alpha, \beta$]{
  ePIE 疊代更新步長參數 (Step-size parameters for ePIE update)
}

\item[$step_1$]{
  初次自動對焦步長 (Initial autofocus step size)
}

\item[$F_{var}$]{
  方差評價函數 (Variance focus metric)
}

\item[$k$]{
  自適應步長比例係數 (Adaptive step scaling coefficient)
}

\item[$\Delta F_0$]{
  判定門檻值 (Decision threshold for AF)
}

\item[$d$]{
  結構週期 / 光柵間距 (Grating period / Structure spacing)
}

\item[$\theta$]{
  繞射角 (Diffraction angle)
}

\item[$m$]{
  繞射階數 (Diffraction order)
}

\item[$f_c$]{
  截止頻率 (Cut-off frequency)
}

\item[$f$]{
  物鏡焦距 (Focal length of objective)
}

\item[$u$]{
  物距 (Object distance)
}

\item[$v$]{
  像距 (Image distance)
}

\item[$U$]{
  色彩解混張量 (Unmixing tensor)
}

\item[$G$]{
  空間增益矩陣 (Gain map for flat-field correction)
}

\end{denotation}
